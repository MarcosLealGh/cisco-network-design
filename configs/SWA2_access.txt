! ============================================================
! CISCO 2960 — SW-A2 (ACCESO — VLAN 30 Operaciones/TI + VLAN 50 Servidores)
! Puertos de acceso con port-security, PortFast y BPDU Guard
!
! Antes de aplicar en el lab, verificar el cableado físico:
!   [ ] Fa0/1 = uplink trunk hacia SW-D (Fa0/2)
!   [ ] Fa0/2-Fa0/15 = hosts VLAN 30 TI (ajustar rango)
!   [ ] Fa0/16-Fa0/20 = servidores (VLAN 50) — IP estática, no DHCP
!   [ ] Fa0/21-Fa0/24 = sin uso
! ============================================================

! ---- IDENTIFICACIÓN ----
hostname SW-A2-ACCESO
enable secret <SET_STRONG_PASSWORD>
service password-encryption
no ip domain-lookup
username admin privilege 15 secret <SET_STRONG_PASSWORD>

line con 0
 logging synchronous
 exec-timeout 30 0
line vty 0 4
 login local
 transport input ssh

! ---- VLANs ----
vlan 30
 name Operaciones-TI
vlan 50
 name Servidores
vlan 99
 name Management
vlan 999
 name UNUSED

! ---- SPANNING TREE ----
spanning-tree mode rapid-pvst

! ---- UPLINK TRUNK HACIA SW-D ----
interface FastEthernet0/1
 description UPLINK_SWD
 switchport mode trunk
 switchport trunk native vlan 99
 switchport trunk allowed vlan 30,50,99
 switchport nonegotiate
 no shutdown

! ---- ACCESO VLAN 30 — Operaciones / TI ----
interface range FastEthernet0/2 - FastEthernet0/15
 description ACCESS_VLAN30_TI
 switchport mode access
 switchport access vlan 30
 spanning-tree portfast
 spanning-tree bpduguard enable
 switchport port-security maximum 1
 switchport port-security violation restrict
 switchport port-security
 no shutdown

! ---- ACCESO VLAN 50 — Servidores ----
! Los servidores usan IP estática (192.168.50.10 helpdesk, 192.168.50.11 SIEM), no DHCP
interface range FastEthernet0/16 - FastEthernet0/20
 description ACCESS_VLAN50_Servidores
 switchport mode access
 switchport access vlan 50
 spanning-tree portfast
 spanning-tree bpduguard enable
 switchport port-security maximum 1
 switchport port-security violation restrict
 switchport port-security
 no shutdown

! ---- PUERTOS SIN USO ----
interface range FastEthernet0/21 - FastEthernet0/24
 description UNUSED
 switchport mode access
 switchport access vlan 999
 shutdown

interface range GigabitEthernet0/1 - GigabitEthernet0/2
 description UNUSED
 shutdown

! ---- MANAGEMENT — SVI VLAN 99 ----
interface Vlan99
 description MANAGEMENT
 ip address 192.168.99.4 255.255.255.0
 no shutdown

ip default-gateway 192.168.99.1

! ---- FIN DE CONFIGURACIÓN ----
end
