! ============================================================
! CISCO 2960 — SW-D (DISTRIBUCIÓN) — raíz de spanning-tree
! Trunks hacia el core y hacia los switches de acceso
!
! Antes de aplicar en el lab, verificar el cableado físico:
!   [ ] Gi0/1 = uplink trunk hacia el core (Gi0/0)
!   [ ] Fa0/1 = trunk hacia SW-A1 (VLAN 10,20 + native 99)
!   [ ] Fa0/2 = trunk hacia SW-A2 (VLAN 30,50 + native 99)
!   [ ] Fa0/3 = trunk hacia SW-A3 (VLAN 10,40 + native 99)
!   [ ] Fa0/4-Fa0/24 = sin uso — VLAN 999 y apagados
! ============================================================

! ---- IDENTIFICACIÓN ----
hostname SW-D-DISTRIBUCION
enable secret <SET_STRONG_PASSWORD>
service password-encryption
no ip domain-lookup
username admin privilege 15 secret <SET_STRONG_PASSWORD>

line con 0
 logging synchronous
 exec-timeout 30 0
line vty 0 4
 login local
 transport input ssh

! ---- VLANs ----
vlan 10
 name Corporativo
vlan 20
 name Ventas
vlan 30
 name Operaciones-TI
vlan 40
 name Invitados-WiFi
vlan 50
 name Servidores
vlan 99
 name Management
vlan 999
 name UNUSED

! ---- SPANNING TREE — este switch es la raíz ----
spanning-tree mode rapid-pvst
spanning-tree vlan 10,20,30,40,50,99 root primary

! ---- UPLINK HACIA EL CORE — trunk todas las VLANs ----
interface GigabitEthernet0/1
 description UPLINK_CORE
 switchport trunk encapsulation dot1q
 switchport mode trunk
 switchport trunk native vlan 99
 switchport trunk allowed vlan 10,20,30,40,50,99
 switchport nonegotiate
 no shutdown

! ---- DOWNLINK HACIA SW-A1 — VLAN 10 y 20 ----
interface FastEthernet0/1
 description TRUNK_SWA1_VLAN10_20
 switchport trunk encapsulation dot1q
 switchport mode trunk
 switchport trunk native vlan 99
 switchport trunk allowed vlan 10,20,99
 switchport nonegotiate
 no shutdown

! ---- DOWNLINK HACIA SW-A2 — VLAN 30 y 50 ----
interface FastEthernet0/2
 description TRUNK_SWA2_VLAN30_50
 switchport trunk encapsulation dot1q
 switchport mode trunk
 switchport trunk native vlan 99
 switchport trunk allowed vlan 30,50,99
 switchport nonegotiate
 no shutdown

! ---- DOWNLINK HACIA SW-A3 — VLAN 10 y 40 (APs) ----
interface FastEthernet0/3
 description TRUNK_SWA3_VLAN10_40_APs
 switchport trunk encapsulation dot1q
 switchport mode trunk
 switchport trunk native vlan 99
 switchport trunk allowed vlan 10,40,99
 switchport nonegotiate
 no shutdown

! ---- PUERTOS SIN USO — aislar y apagar ----
interface range FastEthernet0/4 - FastEthernet0/24
 description UNUSED
 switchport mode access
 switchport access vlan 999
 shutdown

interface GigabitEthernet0/2
 description UNUSED
 shutdown

! ---- MANAGEMENT — SVI VLAN 99 ----
interface Vlan99
 description MANAGEMENT
 ip address 192.168.99.2 255.255.255.0
 no shutdown

ip default-gateway 192.168.99.1

! ---- FIN DE CONFIGURACIÓN ----
end
